```go
func (rs *RaftServer) String() string {
	rs.stateLock.Lock()
	defer rs.stateLock.Unlock()

	return fmt.Sprintf(
		"RaftServer{id=%s, addr=%s, state=%v, currentTerm=%d, votedFor=%s, logLen=%d, commitIndex=%d, lastApplied=%d}",
		rs.id,
		rs.addr,
		rs.state,
		rs.currentTerm.Load(),
		rs.votedFor,
		len(rs.log),
		rs.commitIndex.Load(),
		rs.lastApplied.Load(),
	)
}
```

This locks the state to safely read mutable fields and includes the key identifying and state information without being overly verbose.
